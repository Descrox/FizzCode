Token Types:
  Values:
    INT     - Whole number value
    FLOAT   - Floating point number value
    STR     - String value
    KEY     - Keyword value
    ID      - Identifier value
    
  Operations:
    ADD     - "+" symbol
    SUB     - "-" symbol
    MUL     - "*" symbol
    DIV     - "/" symbol
    EQ      - "=" symbol
    EE      - "==" symbol
    NE      - "!=" symbol
    LT      - "<" symbol
    GT      - ">" symbol
    LTE     - "<=" symbol
    GTE     - ">=" symbol
    NOT     - "!" symbol
    
  Symbols:
    LPAREN  - "(" symbol
    RPAREN  - ")" symbol
    COLON   - ":" symbol
    COMMA   - "," symbol
    LSQUARE - "[" symbol
    RSQUARE - "]" symbol
    
  Special:
    EOF     - End of file
    
Keywords:
  var       - Variable declaration
  and       - Comparison operation
  or        - Comparison operation
  if        - If branch declaration
  elif      - Alternate If branch declaration
  else      - If branch else case declaration
  while     - While loop declaration
  for       - For loop declaration
  
Constant Variables:
  true      - Boolean value
  false     - Boolean value
  null      - Null value
  
Expression Types:
  expr      - Expects any of the following:
              VarAssignNode:
                "var"
                
              IfNode:
                "if"
                
              WhileNode:
                "while"
                
              ForNode:
                "for"
                
              ListNode:
                LSQUARE
                
              BinOpNode:
                "and", "or"
  
  compExpr  - Expects any of the following:
                BinOpNode:
                  EE, NE, LT, GT, LTE, GTE
  
  arithExpr - Expects any of the following:
                BinOpNode:
                  ADD, SUB
  
  term      - Expects any of the following:
                BinOpNode:
                  MUL, DIV
  
  factor    - Expects any of the following:
                UnaryOpNode:
                  SUB, NOT
                
                NumberNode:
                  INT, FLOAT
                
                BinOpNode (parenthesis):
                  LPAREN
                
                VarAccessNode:
                  ID
                
                StringNode:
                  STR
